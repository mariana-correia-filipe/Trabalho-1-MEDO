      # Definir a equao diferencial
      # y'(t) = (0.5 + sin(6t)) * y(t)
      @njit
      def f(t, y):
          return (0.5 + np.sin(6*t)) * y
      # Implementao do mtodo de Euler explcito
      @njit
      def euler(h):
          N = int(40/h) # Nmero de passos
          t = 0.0 # Valor inicial de t
          y = 1.0 # Condio inicial y(0) = 1
          for _ in range(N):
              y = y + h * f(t, y) # Frmula do mtodo de Euler
              t += h # Atualizao de t
          return y # Retorna a aproximao em t = 40
      # Implementao do mtodo de Runge-Kutta de 4 ordem
      @njit
      def rk4(h):
          N = int(40/h) # Nmero de passos
          t = 0.0 # Valor inicial de t
          y = 1.0 # Condio inicial y(0) = 1
          for _ in range(N):
              k1 = f(t, y) # Clculo do primeiro coeficiente k1
              k2 = f(t + h/2, y + h*k1/2) # Clculo do segundo coeficiente k2
              k3 = f(t + h/2, y + h*k2/2) # Clculo do terceiro coeficiente k3
              k4 = f(t + h, y + h*k3) # Clculo do quarto coeficiente k4
              y += (h/6)*(k1 + 2*k2 + 2*k3 + k4) # Frmula do mtodo de Runge-Kutta de 4 ordem
              t += h # Atualizao de t
          return y # Retorna a aproximao em t = 40

      # Aplicar AB4 como preditor
      @njit
      def AB4(h):
          N = int(40/h) # Nmero de passos
          valores_y = np.zeros(4) # Vetor para guardar os ltimos 4 valores de y
          valores_f = np.zeros(4) # Vetor para guardar os ltimos 4 valores de f(t,y)
          t = 0.0 # Valor inicial de t
          valores_y[0] = 1.0 # Condio inicial y(0) = 1
          valores_f[0] = f(t, valores_y[0])
          # Clculo dos valores iniciais atravs do mtodo RK4
          for i in range(1, 4):
              valores_y[i] = rk4_multipasso(t, valores_y[i-1], h)
              t += h
              valores_f[i] = f(t, valores_y[i])
          # Aplicao do mtodo AB4
          for _ in range(3, N):
              novo_y = valores_y[3] + (h/24)*(55*valores_f[3] - 59*valores_f[2] + 37*valores_f[1] - 9*valores_f[0]) # Frmula do Adams-Bashforth de 4 ordem
              t += h
              # Atualizao dos vetores
              valores_y[0], valores_y[1], valores_y[2], valores_y[3] = valores_y[1], valores_y[2], valores_y[3], novo_y
              valores_f[0], valores_f[1], valores_f[2], valores_f[3] = valores_f[1], valores_f[2], valores_f[3], f(t, novo_y)
          return valores_y[3] # Retorna a aproximao em t = 40

      # Aplicar AM4 como corretor (1, 2 e 3 correes)
      @njit
      def AM4(h, k):
          N = int(40/h) # Nmero de passos
          valores_y = np.zeros(4) # Vetor para guardar os ltimos 4 valores de y
          valores_f = np.zeros(4) # Vetor para guardar os ltimos 4 valores de f(t,y)
          t = 0.0 # Valor inicial de t
          valores_y[0] = 1.0 # Condio inicial y(0) = 1
          valores_f[0] = f(t, valores_y[0])
          # Clculo dos valores iniciais atravs do mtodo RK4
          for i in range(1, 4):
              valores_y[i] = rk4_multipasso(t, valores_y[i-1], h)
              t += h
              valores_f[i] = f(t, valores_y[i])
          for _ in range(3, N):
              # Preditor AB4
              preditor_y = valores_y[3] + (h/24)*(55*valores_f[3] - 59*valores_f[2] + 37*valores_f[1] - 9*valores_f[0])
              novo_t = t + h
              # Correes com o mtodo AM4
              corretor_y = preditor_y
              for __ in range(k): # k = nmero de correes
                  corretor_f = f(novo_t, corretor_y)
                  corretor_y = valores_y[3] + (h/24)*(9*corretor_f + 19*valores_f[3] - 5*valores_f[2] + valores_f[1]) # Frmula do Adams-Moulton de 4 ordem
              # Atualizao do valor t
              t = novo_t
              # Atualizao dos vetores
              valores_y[0], valores_y[1], valores_y[2], valores_y[3] = valores_y[1], valores_y[2], valores_y[3], corretor_y
              valores_f[0], valores_f[1], valores_f[2], valores_f[3] = valores_f[1], valores_f[2], valores_f[3], f(t, corretor_y)
          return valores_y[3] # Retorna a aproximao em t = 40
